import 'package:flutter/material.dart';

class StrikeBar extends StatelessWidget {
  final int strikes;
  const StrikeBar({super.key, required this.strikes});

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.symmetric(horizontal: 16),
      child: Container(
        padding: const EdgeInsets.symmetric(horizontal: 14, vertical: 8),
        decoration: BoxDecoration(
          color: const Color(0xFF12121F),
          borderRadius: BorderRadius.circular(12),
          border: Border.all(
            color: strikes >= 2
                ? const Color(0xFFFF3CAC).withOpacity(0.5)
                : Colors.white10,
          ),
        ),
        child: Row(
          children: [
            const Text('STRIKES', style: TextStyle(
              color: Colors.white38, fontSize: 11, fontWeight: FontWeight.w700, letterSpacing: 1,
            )),
            const SizedBox(width: 10),
            Row(
              children: List.generate(3, (i) {
                final active = i < strikes;
                return Padding(
                  padding: const EdgeInsets.only(right: 6),
                  child: AnimatedContainer(
                    duration: const Duration(milliseconds: 300),
                    width: 28,
                    height: 28,
                    decoration: BoxDecoration(
                      shape: BoxShape.circle,
                      color: active
                          ? const Color(0xFFFF3CAC).withOpacity(0.15)
                          : Colors.white08,
                      border: Border.all(
                        color: active
                            ? const Color(0xFFFF3CAC)
                            : Colors.white12,
                        width: active ? 2 : 1,
                      ),
                    ),
                    child: Center(
                      child: Text(
                        '⚡',
                        style: TextStyle(
                          fontSize: 13,
                          color: active ? Colors.white : Colors.white24,
                        ),
                      ),
                    ),
                  ),
                );
              }),
            ),
            const Spacer(),
            if (strikes == 0)
              const Text('Clean 🔥', style: TextStyle(color: Color(0xFF00FF87), fontSize: 11))
            else if (strikes == 1)
              const Text('1 strike', style: TextStyle(color: Colors.orange, fontSize: 11))
            else if (strikes == 2)
              const Text('⚠️ FINAL WARNING', style: TextStyle(color: Color(0xFFFF3CAC), fontSize: 11, fontWeight: FontWeight.bold))
            else
              const Text('LOCKDOWN ACTIVE', style: TextStyle(color: Color(0xFFFF3CAC), fontSize: 11, fontWeight: FontWeight.w900)),
          ],
        ),
      ),
    );
  }
}
